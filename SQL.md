CREATE TABLE metro(
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    hex CHAR(7) NOT NULL,
    path VARCHAR(100) NOT NULL,
    direct boolean NOT NULL DEFAULT 0
);
